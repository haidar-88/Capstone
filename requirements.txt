# -------------------

#

#    - Ubuntu: sudo apt-get install sumo sumo-tools
#    - macOS: brew install sumo
#
# 3. (Optional) Build ns-3 with Python bindings:
#    git clone https://gitlab.com/nsnam/ns-3-dev.git
#    cd ns-3-dev
#    ./waf configure --enable-python --enable-examples
